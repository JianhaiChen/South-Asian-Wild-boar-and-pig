

###################### Estimate the number of novel SNPs and shared SNPs ############
plink -bfile 745sample -freq -within ogornot -out frqogornot --allow-extra-chr

awk '$3=="noog" && $6>0 && $4 ~/[ATCG]/ && $5 ~/[ATCG]/ && length($4)==1 && length($5) ==1 {print $2}'  frqogornot.frq.strat > ingroupmafover0

 plink -bfile 745sample -extract ingroupmafover0 -freq -within ingroup -out ingroupmafover0site --allow-extra-chr   --snps-only &

grep sa ingroupmafover0site.frq.strat |awk '{print $2,$6}' >saallsamplefq&
 awk '$3=="ea"' ingroupmafover0site.frq.strat |awk '{print $6}' >eaallsamplefq&
 grep eu ingroupmafover0site.frq.strat |awk '{print $6}' >euallsamplefq&
 grep isea ingroupmafover0site.frq.strat |awk '{print $6}' >iseaallsamplefq&
 grep msea ingroupmafover0site.frq.strat |awk '{print $6}' >mseaallsamplefq&

#### get the number of all SNPs within each populations
awk '$1>0' eaallsamplefq |wc -l > eaallsamplefq.stat &
awk '$1>0' euallsamplefq |wc -l > euallsamplefq.stat &
awk '$2>0' saallsamplefq |wc -l  > saallsamplefq.stat &
awk '$1>0' iseaallsamplefq |wc -l > iseaallsamplefq.stat &
awk '$1>0' mseaallsamplefq |wc -l > mseaallsamplefq.stat &

#### how many SNPs in non-South Asian pops could be found in South Asian pop?

#The number of common SNPs in East Asian pop that could also be found in South Asian pop
paste saallsamplefq iseaallsamplefq mseaallsamplefq euallsamplefq eaallsamplefq |awk '$2>0 && $6>0.05'|wc -l

#The number of common SNPs in European pop that could also be found in South Asian pop
 paste saallsamplefq iseaallsamplefq mseaallsamplefq euallsamplefq eaallsamplefq |awk '$2>0 && $5>0.05'|wc -l

#The number of common SNPs in MSEA pop that could also be found in South Asian pop
 paste saallsamplefq iseaallsamplefq mseaallsamplefq euallsamplefq eaallsamplefq |awk '$2>0 && $4>0.05'|wc -l

#The number of common SNPs in ISEA pop that could also be found in South Asian pop
 paste saallsamplefq iseaallsamplefq mseaallsamplefq euallsamplefq eaallsamplefq |awk '$2>0 && $3>0.05'|wc -l
